#### **7.1.2 Canvas Initialization - High-Density Grid Setup**
```javascript
class CanvasManager {
    constructor(canvasId) {
        this.canvas = document.getElementById(canvasId);
        this.ctx = this.canvas.getContext('2d');
        this.setupHighDensityCanvas();
        this.initializeEventHandlers();
    }

    setupHighDensityCanvas() {
        // HIGH-DENSITY CONFIGURATION: 320x240 grid = 76,800 cells (100x denser than 96x96)
        this.G_X = 320;          // Grid width - 3.3x wider than standard
        this.G_Y = 240;          // Grid height - 2.5x taller than standard
        this.CELL_SIZE_X = 5;    // 5px per cell horizontally
        this.CELL_SIZE_Y = 5;    // 5px per cell vertically
        this.W = 1600;           // Total width: 1600px (high resolution)
        this.H = 1200;           // Total height: 1200px (high resolution)

        // Set canvas to high-density dimensions
        this.canvas.width = this.W;
        this.canvas.height = this.H;

        // CSS scales to fit container while maintaining aspect ratio
        this.canvas.style.width = '100%';
        this.canvas.style.height = 'auto';
        this.canvas.style.maxWidth = this.W + 'px';
        this.canvas.style.maxHeight = this.H + 'px';
    }

    initializeEventHandlers() {
        // IMMEDIATE TOUCH RESPONSE: Direct event listeners for instant interaction
        this.setupPointerEvents();
        this.setupTouchEvents();
        this.setupGestureEvents();
    }

    // Convert screen coordinates to grid coordinates for high-density interaction
    screenToGrid(screenX, screenY) {
        const rect = this.canvas.getBoundingClientRect();
        const scaleX = this.canvas.width / rect.width;
        const scaleY = this.canvas.height / rect.height;

        const canvasX = (screenX - rect.left) * scaleX;
        const canvasY = (screenY - rect.top) * scaleY;

        const gridX = Math.floor(canvasX / this.CELL_SIZE_X);
        const gridY = Math.floor(canvasY / this.CELL_SIZE_Y);

        return {
            gridX: Math.max(0, Math.min(this.G_X - 1, gridX)),
            gridY: Math.max(0, Math.min(this.G_Y - 1, gridY))
        };
    }
}
```

```javascript
    constructor(config) {
        this.G_X = config.G_X;
        this.G_Y = config.G_Y;
        this.CELL_SIZE_X = config.CELL_SIZE_X;
        this.CELL_SIZE_Y = config.CELL_SIZE_Y;

    // Convert screen coordinates to high-density grid coordinates
    screenToGrid(screenX, screenY) {
        const rect = canvas.getBoundingClientRect();
        const scaleX = canvas.width / rect.width;
        const scaleY = canvas.height / rect.height;

        const canvasX = (screenX - rect.left) * scaleX;
        const canvasY = (screenY - rect.top) * scaleY;

        return {
            gridX: Math.floor(canvasX / this.CELL_SIZE_X),
            gridY: Math.floor(canvasY / this.CELL_SIZE_Y)
        };
    }

    // High-density paint function with larger brush
    paintAt(x, y, mode) {
        const { gridX, gridY } = this.screenToGrid(x, y);
        const brushSize = HIGH_DENSITY_CONFIG.BRUSH_SIZE;

        for (let dy = -brushSize; dy <= brushSize; dy++) {
            for (let dx = -brushSize; dx <= brushSize; dx++) {
                const nx = gridX + dx;
                const ny = gridY + dy;

                if (nx >= 0 && nx < this.G_X && ny >= 0 && ny < this.G_Y) {
                    const index = ny * this.G_X + nx;
                    this.applyBrushEffect(index, mode);
                }
            }
        }
```

```javascript
class EpistemicLedger {
    constructor(simulationName) {
        this.simulationName = simulationName;
        this.events = [];
        this.hashChain = [];
        this.keyPair = null;
        this.initializeCryptography();
    }

    async initializeCryptography() {
        // Generate ECDSA P-256 key pair
        this.keyPair = await crypto.subtle.generateKey(
            {
                name: "ECDSA",
                namedCurve: "P-256"
            },
            true,
            ["sign", "verify"]
        );
    }

    recordEvent(eventType, data, timestamp = null) {
        const event = {
            t: timestamp || this.getCurrentTime(),
            type: eventType,
            data: this.canonicalizeData(data),
            hash: null
        };
        
        // Add to chain with previous hash
        event.hash = this.computeEventHash(event);
        
        this.events.push(event);
        this.updateHashChain(event);
        
        return event;
    }

    canonicalizeData(data) {
        // Ensure consistent JSON serialization
        return JSON.parse(JSON.stringify(data, Object.keys(data).sort()));
    }

    computeEventHash(event) {
        const eventString = JSON.stringify(event, Object.keys(event).sort());
        const encoder = new TextEncoder();
        const data = encoder.encode(eventString);
        
        return crypto.subtle.digest('SHA-256', data)
            .then(hash => {
                const hashArray = Array.from(new Uint8Array(hash));
                return hashArray.map(b => b.toString(16).padStart(2, '0')).join('');
            });
    }

    async signLedger() {
        const ledgerContent = JSON.stringify({
            metadata: this.generateMetadata(),
            events: this.events
        });
        
        const encoder = new TextEncoder();
        const data = encoder.encode(ledgerContent);
        
        const signature = await crypto.subtle.sign(
            {
                name: "ECDSA",
                hash: { name: "SHA-256" }
            },
            this.keyPair.privateKey,
            data
        );
        
        return {
            ledger: ledgerContent,
            signature: btoa(String.fromCharCode(...new Uint8Array(signature))),
            publicKey: await this.exportPublicKey()
        };
    }
}
```

```javascript
class LedgerVerifier {
    static async verifyLedger(ledgerData, signature, publicKey) {
        try {
            // Import public key
            const key = await crypto.subtle.importKey(
                'raw',
                Uint8Array.from(atob(publicKey), c => c.charCodeAt(0)),
                {
                    name: "ECDSA",
                    namedCurve: "P-256"
                },
                false,
                ["verify"]
            );
            
            // Verify signature
            const encoder = new TextEncoder();
            const data = encoder.encode(ledgerData);
            const signatureBytes = Uint8Array.from(atob(signature), c => c.charCodeAt(0));
            
            const isValid = await crypto.subtle.verify(
                {
                    name: "ECDSA",
                    hash: { name: "SHA-256" }
                },
                key,
                signatureBytes,
                data
            );
            
            if (!isValid) {
                throw new Error('Signature verification failed');
            }
            
            // Verify hash chain integrity
            return this.verifyHashChain(JSON.parse(ledgerData));
            
        } catch (error) {
            console.error('Ledger verification failed:', error);
            return false;
        }
    }

    static verifyHashChain(ledger) {
        let previousHash = '';
        
        for (const event of ledger.events) {
            const computedHash = this.computeEventHash(event);
            if (event.hash !== computedHash) {
                throw new Error(`Hash chain broken at event ${event.t}`);
            }
            
            // Verify chain linkage
            if (previousHash && !event.hash.includes(previousHash.substring(0, 8))) {
                throw new Error(`Hash chain linkage broken at event ${event.t}`);
            }
            
            previousHash = event.hash;
        }
        
        return true;
    }
}
```

```javascript
class StatisticalValidator {
    static calculateConfidenceInterval(samples, confidence = 0.95) {
        const n = samples.length;
        if (n < 2) return { mean: samples[0], ci: [samples[0], samples[0]] };
        
        const mean = samples.reduce((a, b) => a + b) / n;
        const variance = samples.reduce((a, b) => a + (b - mean) ** 2, 0) / (n - 1);
        const stdDev = Math.sqrt(variance);
        
        // t-distribution approximation for n < 30
        const tValue = n >= 30 ? 1.96 : this.getTValue(n, confidence);
        const margin = tValue * stdDev / Math.sqrt(n);
        
        return {
            mean: mean,
            ci: [mean - margin, mean + margin],
            stdDev: stdDev,
            n: n
        };
    }
    
    static getTValue(n, confidence) {
        // Approximation of t-distribution critical values
        const df = n - 1;
        if (df <= 30) {
            const tTable = {
                1: 12.706, 2: 4.303, 3: 3.182, 4: 2.776, 5: 2.571,
                6: 2.447, 7: 2.365, 8: 2.306, 9: 2.262, 10: 2.228
            };
            return tTable[Math.min(df, 10)] || 2.0;
        }
        return 1.96; // Normal approximation
    }
}
```

```javascript
class CustomKinematicAgent extends KinematicAgent {
    constructor(config) {
        super(config);
        
        // Define kinematic structure
        this.nodes = this.createNodes(config);
        this.connections = this.createConnections(config);
        
        // Initialize state
        this.initializeState(config);
    }
    
    createNodes(config) {
        return [
            { id: 'node1', position: [0, 0], velocity: [0, 0], mass: 1.0 },
            { id: 'node2', position: [1, 0], velocity: [0, 0], mass: 1.0 }
        ];
    }
    
    createConnections(config) {
        return [
            { 
                u: 'node1', 
                v: 'node2', 
                kind: 'spring', 
                stiffness: config.stiffness || 100,
                damping: config.damping || 0.1,
                restLength: 1.0
            }
        ];
    }
    
    step(dt) {
        // Apply forces
        this.applySpringForces(dt);
        this.applyDampingForces(dt);
        
        // Integrate motion
        this.integrateVerlet(dt);
        
        // Update visual representation
        this.updateVisualPositions();
        
        super.step(dt);
    }
    
    applySpringForces(dt) {
        this.connections.forEach(conn => {
            if (conn.kind === 'spring') {
                const nodeU = this.nodes.find(n => n.id === conn.u);
                const nodeV = this.nodes.find(n => n.id === conn.v);
                
                const displacement = [
                    nodeV.position[0] - nodeU.position[0],
                    nodeV.position[1] - nodeU.position[1]
                ];
                
                const distance = Math.sqrt(displacement[0]**2 + displacement[1]**2);
                const forceMagnitude = conn.stiffness * (distance - conn.restLength);
                
                const force = [
                    forceMagnitude * displacement[0] / distance,
                    forceMagnitude * displacement[1] / distance
                ];
                
                // Apply equal and opposite forces
                nodeU.force = nodeU.force || [0, 0];
                nodeV.force = nodeV.force || [0, 0];
                
                nodeU.force[0] += force[0];
                nodeU.force[1] += force[1];
                nodeV.force[0] -= force[0];
                nodeV.force[1] -= force[1];
            }
        });
    }
    
    integrateVerlet(dt) {
        this.nodes.forEach(node => {
            const acceleration = [
                node.force[0] / node.mass,
                node.force[1] / node.mass
            ];
            
            // Verlet integration for stability
            const newPosition = [
                2 * node.position[0] - node.prevPosition[0] + acceleration[0] * dt * dt,
                2 * node.position[1] - node.prevPosition[1] + acceleration[1] * dt * dt
            ];
            
            node.prevPosition = node.position;
            node.position = newPosition;
            node.velocity = [
                (newPosition[0] - node.prevPosition[0]) / dt,
                (newPosition[1] - node.prevPosition[1]) / dt
            ];
            
            // Reset forces
            node.force = [0, 0];
        });
    }
}
```

        this.imageMobject.set_opacity(this.opacity);

        // Convert simulation state to RGB image data
        // This depends on the specific simulation type
        
        if (state.type === 'scalar_field') {
            return this.scalarFieldToRGB(state.data, state.min, state.max);
        } else if (state.type === 'vector_field') {
            return this.vectorFieldToRGB(state.data);
        } else if (state.type === 'particle_system') {
            return this.particleSystemToRGB(state.particles);
        }
        
        // Default: return blank image
        return new Uint8Array(4 * this.width * this.height);
    }
    
    scalarFieldToRGB(field, minVal, maxVal) {
        const width = field[0].length;
        const height = field.length;
        const rgb = new Uint8Array(4 * width * height);
        
        for (let y = 0; y < height; y++) {
            for (let x = 0; x < width; x++) {
                const normalizedValue = (field[y][x] - minVal) / (maxVal - minVal);
                const color = this.getColormapColor(normalizedValue);
                
                const index = 4 * (y * width + x);
                rgb[index] = color.r;
                rgb[index + 1] = color.g;
                rgb[index + 2] = color.b;
                rgb[index + 3] = 255; // Alpha
            }
        }
        
        return rgb;
    }
    
    getColormapColor(value) {
        // Simple grayscale colormap - override for custom colors
        const intensity = Math.max(0, Math.min(255, value * 255));

```javascript
        element.addEventListener(event, handler);
        if (!this.listeners.has(element)) {
            this.listeners.set(element, []);
```

```javascript
// Gauss-Seidel Successive Over-Relaxation (SOR)
function solveNetworkGS_SOR(G, b, omega = 1.2) {
    const n = G.length;
    let v = new Array(n).fill(0);
    let v_new = [...v];
    let iterations = 0;
    const maxIter = 1000;
    const tolerance = 1e-6;

    while (iterations < maxIter) {
        let maxDiff = 0;

        for (let i = 0; i < n; i++) {
            let sum = b[i];

            // Subtract known voltages
            for (let j = 0; j < n; j++) {
                if (i !== j) {
                    sum -= G[i][j] * v[j];
                }
            }

            // Update with over-relaxation
            const new_v = (1 - omega) * v[i] + omega * sum / G[i][i];
            const diff = Math.abs(new_v - v[i]);
            maxDiff = Math.max(maxDiff, diff);

            v_new[i] = new_v;
        }

        // Swap arrays
        [v, v_new] = [v_new, v];
        iterations++;

        if (maxDiff < tolerance) break;
    }

    return { voltages: v, iterations, converged: iterations < maxIter };
}
```

```javascript
// Dijkstra's Algorithm for Optimal Path
function findOptimalPath(graph, start, end) {
    const distances = {};
    const previous = {};
    const queue = new PriorityQueue();

    // Initialize
    for (let node in graph.nodes) {
        distances[node] = Infinity;
        previous[node] = null;
    }
    distances[start] = 0;
    queue.enqueue(start, 0);

    while (!queue.isEmpty()) {
        const current = queue.dequeue();

        for (let neighbor in graph.nodes[current].connections) {
            const resistance = graph.edges[current][neighbor].resistance;
            const alt = distances[current] + resistance;

            if (alt < distances[neighbor]) {
                distances[neighbor] = alt;
                previous[neighbor] = current;
                queue.enqueue(neighbor, alt);
            }
        }
    }

    return reconstructPath(previous, end);
}
```

```javascript
    const u_new = new Array(u.length);

    for (let i = 0; i < u.length; i++) {
        for (let j = 0; j < u[i].length; j++) {

            const lapU = laplacian(u, i, j);
            const lapV = laplacian(v, i, j);

            // Reaction terms

        }
    }

}
```

```javascript
function applyVorticityConfinement(u, v, vortConf) {
    // 1. Compute vorticity ω = ∂v/∂x - ∂u/∂y
    for (let i = 1; i < NX-1; i++) {
        for (let j = 1; j < NY-1; j++) {
            const dv_dx = (v[idx(i+1,j)] - v[idx(i-1,j)]) / (2*dx);
            const du_dy = (u[idx(i,j+1)] - u[idx(i,j-1)]) / (2*dy);
            vorticity[idx(i,j)] = dv_dx - du_dy;
        }
    }

    // 2. Compute confinement force
    for (let i = 1; i < NX-1; i++) {
        for (let j = 1; j < NY-1; j++) {
            // Gradient of |ω|
            const dw_dx = (Math.abs(vorticity[idx(i+1,j)]) - Math.abs(vorticity[idx(i-1,j)])) / (2*dx);
            const dw_dy = (Math.abs(vorticity[idx(i,j+1)]) - Math.abs(vorticity[idx(i,j-1)])) / (2*dy);

            const length = Math.sqrt(dw_dx*dw_dx + dw_dy*dw_dy) + 1e-6;
            const Nx = dw_dx / length;
            const Ny = dw_dy / length;

            // Perpendicular force: (-Ny, Nx) or (Ny, -Nx)
            const forceScale = vortConf * vorticity[idx(i,j)];
            u[idx(i,j)] += forceScale * (-Ny) * dt;
            v[idx(i,j)] += forceScale * Nx * dt;
        }
    }
}
```

```javascript
// Gauss-Seidel Iterative Method
function gaussSeidelSolve(phi, fixed, target, maxIter = 1000) {
    let iteration = 0;
    let residual = Infinity;

    while (iteration < maxIter && residual > tolerance) {
        residual = 0;

        for (let j = 1; j < NY-1; j++) {
            for (let i = 1; i < NX-1; i++) {
                const idx = j * NX + i;

                // Skip fixed boundary points
                if (fixed[idx]) continue;

                // 5-point Laplacian stencil
                const center = phi[idx];
                const left = phi[idx - 1];
                const right = phi[idx + 1];
                const up = phi[idx - NX];
                const down = phi[idx + NX];

                // Gauss-Seidel update
                const newValue = (left + right + up + down) / 4.0;
                const diff = newValue - center;

                phi[idx] = newValue;
                residual += diff * diff;
            }
        }

        residual = Math.sqrt(residual / (NX * NY));
        iteration++;
    }

    return { iterations: iteration, residual: residual };
}

// Successive Over-Relaxation (SOR)
function sorSolve(phi, fixed, target, omega = 1.85) {
    let iteration = 0;
    let residual = Infinity;

    while (iteration < maxIter && residual > tolerance) {
        residual = 0;

        for (let j = 1; j < NY-1; j++) {
            for (let i = 1; i < NX-1; i++) {
                const idx = j * NX + i;

                if (fixed[idx]) continue;

                const center = phi[idx];
                const left = phi[idx - 1];
                const right = phi[idx + 1];
                const up = phi[idx - NX];
                const down = phi[idx + NX];

                // SOR update with over-relaxation
                const gaussValue = (left + right + up + down) / 4.0;
                const newValue = center + omega * (gaussValue - center);

                const diff = newValue - center;
                phi[idx] = newValue;
                residual += diff * diff;
            }
        }

        residual = Math.sqrt(residual / (NX * NY));
        iteration++;
    }

    return { iterations: iteration, residual: residual };
}
```

```javascript
function traceEquipotential(phi, targetValue, startX, startY) {
    const points = [[startX, startY]];
    const visited = new Set();

    for (let step = 0; step < maxContourSteps; step++) {
        const current = points[points.length - 1];
        const i = Math.floor(current[0]);
        const j = Math.floor(current[1]);

        if (i < 0 || i >= NX-1 || j < 0 || j >= NY-1) break;

        // Find direction perpendicular to gradient
        const gradX = interpolateGradientX(phi, current[0], current[1]);
        const gradY = interpolateGradientY(phi, current[0], current[1]);

        // Perpendicular direction for contour following
        const perpX = -gradY;
        const perpY = gradX;

        const magnitude = Math.sqrt(perpX*perpX + perpY*perpY);
        if (magnitude < 1e-6) break;

        const nx = perpX / magnitude;
        const ny = perpY / magnitude;

        // Try both directions
        const candidates = [
            [current[0] + contourStep * nx, current[1] + contourStep * ny],
            [current[0] - contourStep * nx, current[1] - contourStep * ny]
        ];

        let bestCandidate = null;
        let bestError = Infinity;

        for (const candidate of candidates) {
            const ci = Math.floor(candidate[0]);
            const cj = Math.floor(candidate[1]);

            if (ci < 0 || ci >= NX-1 || cj < 0 || cj >= NY-1) continue;

            const interpolated = bilinearInterpolate(phi, candidate[0], candidate[1]);
            const error = Math.abs(interpolated - targetValue);

            if (error < bestError) {
                bestError = error;
                bestCandidate = candidate;
            }
        }

        if (!bestCandidate || bestError > contourTolerance) break;

        const key = `${Math.round(bestCandidate[0]*10)},${Math.round(bestCandidate[1]*10)}`;
        if (visited.has(key)) break; // Loop detected

        visited.add(key);
        points.push(bestCandidate);
    }

    return points;
}
```

```javascript
// Explicit FDTD scheme
function fdtdStep(u_prev, u_now, u_next, c, dt, dx, geometry) {
    const CFL = c * dt / dx;

    if (geometry === 'String1D') {
        // 1D string with fixed boundaries
        for (let i = 1; i < NX-1; i++) {
            const laplacian = u_now[i-1] - 2*u_now[i] + u_now[i+1];
            u_next[i] = 2*u_now[i] - u_prev[i] + CFL*CFL * laplacian;
        }
        // Boundary conditions
        u_next[0] = 0; u_next[NX-1] = 0;

    } else if (geometry === 'Square2D') {
        // 2D plate with fixed boundaries
        for (let j = 1; j < NY-1; j++) {
            for (let i = 1; i < NX-1; i++) {
                const idx = j * NX + i;
                const laplacian =
                    u_now[idx-1] + u_now[idx+1] +           // x-direction
                    u_now[idx-NX] + u_now[idx+NX] -         // y-direction
                    4*u_now[idx];                           // center

                u_next[idx] = 2*u_now[idx] - u_prev[idx] + CFL*CFL * laplacian;
            }
        }
        // Fixed boundaries
        for (let i = 0; i < NX; i++) {
            u_next[i] = 0; u_next[(NY-1)*NX + i] = 0; // top/bottom
        }
        for (let j = 0; j < NY; j++) {
            u_next[j*NX] = 0; u_next[j*NX + NX-1] = 0; // left/right
        }

    } else if (geometry === 'Circle2D') {
        // Circular membrane with free boundaries
        const R = Math.min(NX, NY) * 0.48;
        const cx = (NX - 1) / 2, cy = (NY - 1) / 2;

        for (let j = 1; j < NY-1; j++) {
            for (let i = 1; i < NX-1; i++) {
                const r = Math.hypot(i - cx, j - cy);
                if (r >= R) continue; // Outside circle

                const idx = j * NX + i;
                const laplacian =
                    u_now[idx-1] + u_now[idx+1] +
                    u_now[idx-NX] + u_now[idx+NX] -
                    4*u_now[idx];

                u_next[idx] = 2*u_now[idx] - u_prev[idx] + CFL*CFL * laplacian;
            }
        }
    }
}
```

```javascript
function computeSpectrogram(u, specBuf, specPtr, N) {
    // Copy current displacement to spectrogram buffer
    for (let i = 0; i < NX; i++) {
        specBuf[specPtr] = u[i]; // For 1D string
        specPtr = (specPtr + 1) % N;
    }

    // Compute DFT magnitude
    const mags = new Float32Array(N / 2);
    for (let k = 0; k < N / 2; k++) {
        let re = 0, im = 0;
        for (let n = 0; n < N; n++) {
            // Hamming window for spectral leakage reduction
            const window = 0.54 - 0.46 * Math.cos(2 * Math.PI * n / (N - 1));
            const angle = -2 * Math.PI * k * n / N;

            re += specBuf[(specPtr - n + N) % N] * window * Math.cos(angle);
            im += specBuf[(specPtr - n + N) % N] * window * Math.sin(angle);
        }
        mags[k] = Math.sqrt(re * re + im * im);
    }

    return mags;
}
```

```javascript
            const curvature = u_now[i-1] - 2*u_now[i] + u_now[i+1];
            potentialEnergy += 0.5 * curvature * curvature;

                    u_now[idx-1] + u_now[idx+1] +
                    u_now[idx-NX] + u_now[idx+NX] -
                    4*u_now[idx];

                potentialEnergy += 0.5 * laplacian * laplacian;
            }
        }
    }
```

```javascript
function shallowWaterStep(h, u, v, dt, dx, dy, g = 9.81) {
    const NX = h[0].length;
    const NY = h.length;

    // Create copies for updates
    const h_new = h.map(row => [...row]);
    const u_new = u.map(row => [...row]);
    const v_new = v.map(row => [...row]);

    // 1. Advect height field
    for (let j = 1; j < NY-1; j++) {
        for (let i = 1; i < NX-1; i++) {
            // Semi-Lagrangian advection for height
            const x_back = i - dt * u[j][i] / dx;
            const y_back = j - dt * v[j][i] / dy;

            // Bilinear interpolation
            h_new[j][i] = bilinearInterpolate(h, x_back, y_back);
        }
    }

    // 2. Update velocities (simplified Euler)
    for (let j = 1; j < NY-1; j++) {
        for (let i = 1; i < NX-1; i++) {
            // Pressure gradient terms
            const dh_dx = (h[j][i+1] - h[j][i-1]) / (2 * dx);
            const dh_dy = (h[j+1][i] - h[j-1][i]) / (2 * dy);

            // Advection terms (simplified)
            const u_adv = -u[j][i] * (u[j][i+1] - u[j][i-1]) / (2 * dx);
            const v_adv = -v[j][i] * (v[j+1][i] - v[j-1][i]) / (2 * dy);

            // Update velocities
            u_new[j][i] = u[j][i] + dt * (u_adv - g * dh_dx);
            v_new[j][i] = v[j][i] + dt * (v_adv - g * dh_dy);
        }
    }

    return { h: h_new, u: u_new, v: v_new };
}
```

```javascript
                    v[j][i] += radialVelocity * (dy / distance);

            }
        }
    }
}
```

## **29. Lessons Learned: Implementation Insights & Critical Fixes**

### **29.1 Cryptographic Consistency Crisis**

**Major Discovery:** During signature verification implementation, we uncovered a critical **cryptographic inconsistency** across simulations:

**The Problem:**
- Fire Percolation: Used ECDSA with JWK keys
- Memristor Pathfinder: Used ECDSA with JWK keys
- Morphogen Lab: Used Ed25519 with raw keys
- Other simulations: Used Ed25519 with raw keys

**Result:** Fire Percolation signatures failed verification while other simulations worked perfectly.

**Solution Implemented:**
1. **Standardized on Ed25519** across ALL simulations for uniformity
2. **Unified key format** to raw binary (32 bytes) for Ed25519
3. **Consistent signature generation** pattern across the entire platform

**Lesson:** Never assume cryptographic consistency - **explicitly standardize** cryptographic approaches across all components.

### **29.2 Signature Verification Debugging Protocol**

**Critical Finding:** Signature verification failures are often caused by subtle data mismatches:

**Common Failure Points:**
1. **Missing event_root**: Fire Percolation initially excluded `event_root` from signing data
2. **Canonicalization mismatch**: Signing used canonicalization but verification didn't
3. **Key format inconsistency**: Different simulations used different key export formats
4. **Algorithm differences**: Mixed use of ECDSA vs Ed25519 across platform

**Debugging Protocol:**
```javascript
// 1. Verify signature algorithm consistency
console.log('Algorithm:', metaData.sig_alg); // Should be 'ed25519' for all

// 2. Check canonicalization match
const signingData = JSON.stringify(canonicalize(metaToSign));
const verificationData = JSON.stringify(canonicalize(metaMinusSig));

// 3. Ensure event_root inclusion
if (!metaToSign.event_root) {
    console.error('MISSING event_root in signing data!');
}
```

**Lesson:** Implement **comprehensive signature debugging** protocols for all cryptographic operations.
